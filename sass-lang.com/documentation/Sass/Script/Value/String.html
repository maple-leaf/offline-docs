def to_s(opts = {})
  if @type == :identifier
    return @value.gsub(/\n\s*/, " ")
  end

  return "\"#{value.gsub('"', "\\\"")}\"" if opts[:quote] == %q{"}
  return "'#{value.gsub("'", "\\'")}'" if opts[:quote] == %q{'}
  return "\"#{value}\"" unless value.include?('"')
  return "'#{value}'" unless value.include?("'")
  "\"#{value.gsub('"', "\\\"")}\"" # '
end
