.example {
  #{elements-of-type(block)} {
    border: 1px solid #777777;
    margin: 1em 3em;
  }
  #{elements-of-type(inline)} {
    color: #cc0000;
  }
}

.example
  #{elements-of-type(block)}
    border: 1px solid #777
    margin: 1em 3em
  #{elements-of-type(inline)}
    color: #c00
